function getMessage() {

    let msg = document.getElementById("txtMessage").value;

    //check if the uesr entered a string
    if (msg == "") {
        Swal.fire("Oops!","Please enter text.");
    } else {

        msg = msg.toUpperCase();

        const regex = /[^a-z0-9]/gi;
        msg = msg.replace(regex, "");

        let palinObj = reverseMessageB(msg);
        displayMessage(palinObj);
    }

}

function reverseMessageA(msg) {
    let revMessage = "";

    //decrimenting loop
    for (let i = msg.length - 1; i >= 0; i--) {

        let char = msg[i];
        revMessage += char;

    }

    return revMessage;
}

function reverseMessageB(originalWord) {

    let revWord = "";

    let returnObj = {};

    revWord = originalWord.split("").reverse().join("");

    if (revWord == originalWord) {

        returnObj.IsPalindrome = true;
    } else {
        returnObj.message = `The word/phrase is not a palindrome :(`;
        returnObj.IsPalindrome = false;
    }

    returnObj.originalWord = originalWord;

function displayMessage(palinObj) {

The code is structured in three functions.

getMessage()

Gets the values entered by the user and stores it into the variable msg. Then tests to make sure the msg is not an empty string and if not, changes the string to upper case. Then a new variable revMessage is declared by calling the function reverseMessage(). Lastly, the displayMessage() is called.

reverseMessageB()

An empty variable revMessage is declared. Then all spaces or special characters are removed from the original input msg using a regex expression. The msg variable is then converted to an array, reversed and joined back into a string. This new reversed string is then assigned to the revMessage variable. Now if else statements are used to test if the input is a palindrome by comparing the joined msg variable to the newly assigned revMessage variable and returning the variable revMessage from the function in both cases, as well as if the original input was truly a palindrome.

displayMessage()

The variable element is assigned to the "results" class within the HTML section where the returned values from the function reverseMessage will be placed. Then the variable revMessage that was passed into the displayMessage will be assigned to element.innerHTML to store it within the HTML section indicated.
